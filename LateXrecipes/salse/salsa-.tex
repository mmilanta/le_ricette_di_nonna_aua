\section{Salsa }
\subsection{Ingredienti}
\begin{itemize}
\item Olio  
\item Limone  
\item Latte o panna  
\item Senape  
\item Pezzetti di gorgonzola  
\item Noci  
\item Pinoli  
\item Sale
\end{itemize}
\subsection{Procedimento}
\begin{enumerate}
\item  Mettere tutti gli ingredienti in un vasetto e agitare.
\end{enumerate}
