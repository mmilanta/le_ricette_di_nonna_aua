\section{Guacamole}
\subsection{Ingredienti}
\begin{itemize}
\item 2 grossi avocado maturi  
\item Coriandolo in polvere  
\item Peperoncino piccante fresco  
\item Tabasco  
\item Lime  
\item Aglio  
\item Cipolla  
\item Sale
\end{itemize}
